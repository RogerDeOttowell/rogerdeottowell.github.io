    #include<stdarg.h>

void my_printf(const char* format, ...)
{
  va_lists arg;
  va_start(arg, format);

  while(*format)
  {
      if(*format == '%')
      {
          format++;
          switch(*format)
          {
              case'd':
              {
                  int num = va_arg(arg, int);
                  print_int(num);
                  break;
              }
              case's':
              {
                  char* str = vaarg(arg, char*);
                  print_str(str);
                  break;
              }
          }
      }
      else
      {
          putchar(*format);
      }
      format++;
  }

  va_end(arg);
}

int add_multiple_values(int argcount, ...)
{
  int counter, res = 0;
  va_list arg;
  va_start(arg, argcount);
  for(counter=0; counter<argcount, counter++)
  {
      res += va_arg(arg. int);
  }
  va_end(arg);
  return res;
}
