inline int Npl(Heap* x)
{
    return x ? x->Npl : -1;
}

Heap* Merge(Heap* H1, Heap* H2)
{
    if(!H1)
        return H2;
    if(!H2)
        return H1;
    
    Heap *a = H1, *b = H2;
    vector<Heap*> stk;
    while (a && b)
    {
        if(a->val > b->val)
            swap(a, b);
            stk.push_back(a);
            a = a->right;
    }

    Heap* tail = a ? a : b;

    while(!stk.empty())
    {
        Heap* p = stk.back();
        stk.pop_back();
        p->right = tail;
        if(Npl(p->left) < Npl(p->right))
            swap(p->left, p->right);
        p->Npl = Npl(p->right) + 1;
        tail = p;
    }

    return tail;
}
